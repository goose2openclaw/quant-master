```bash
git clone https://github.com/goose2openclaw/openclaw-workspace.git
cd openclaw-workspace

# 运行设置
bash gg_start.sh
```

### 使用
```bash
# 启动自主交易
bash scripts/gg_autonomous_trader.sh

# 查看帮助
cat scripts/gg_autonomous_trader.sh --help
```
